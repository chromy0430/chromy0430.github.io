164 private void UpdateTime()
165 {
166 // 이전 시간 값 저장
167 int prevHour = CurrentHour;
168 int prevMinute = CurrentMinute;
169 int prevDay = CurrentDay;
170
171 // 시간 업데이트
172 currentTime += Time.deltaTime * timeMultiplier;
173 if (currentTime >= 86400f) // 하루는 86400초 (24시간)
174 {
175 currentTime -= 86400f;
176 currentDay++; // 하루가 지나면 일차 증가
177 CurrentDay = currentDay;
178 }
179
180 // 시간 값 업데이트
181 UpdateTimeValues();
182
183
184
185 // 시간 변경 이벤트 발생
186 if (CurrentHour != prevHour)

190 }
