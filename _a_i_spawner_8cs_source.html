145 private void CheckForSpawn(int hour, int minute)
146 {
147 // 정각(0분)일 때만 체크하고, 이미 이 시간에 스폰했다면 건너뛰기
148 if (minute == 0 && spawnTimes.Contains(hour) && lastSpawnHour != hour)
149 {
150 int spawnCount = Random.Range(minSpawner, maxSpawner + 1);

152 lastSpawnHour = hour;
153
154 DebugLog($"{hour}시 정각: {spawnCount}명의 AI 스폰 (범위: {minSpawner}~{maxSpawner})", true);
155 }

280 public float GetNextSpawnTime()
281 {
282 if (timeSystem == null || spawnTimes.Count == 0) return 0f;
283
284 int currentHour = timeSystem.CurrentHour;
285 int currentMinute = timeSystem.CurrentMinute;
286 int currentTimeInMinutes = currentHour * 60 + currentMinute;
287
288 // 오늘 남은 스폰 시간 찾기
289 foreach (int spawnHour in spawnTimes)
290 {
291 int spawnTimeInMinutes = spawnHour * 60;
292 if (spawnTimeInMinutes > currentTimeInMinutes)
293 {
294 return spawnTimeInMinutes;
295 }
296 }
297
298 // 오늘 남은 스폰 시간이 없으면 내일 첫 번째 스폰 시간
299 if (spawnTimes.Count > 0)
300 {
301 return spawnTimes[0] * 60 + 1440; // 1440분 = 24시간
302 }
303
304 return 0f;
