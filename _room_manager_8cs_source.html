72 {
73 allRooms.Clear();

75 // 태그로 방 찾기
76 GameObject[] roomObjects = GameObject.FindGameObjectsWithTag(roomTag);

78 {
79 RoomContents roomContents = roomObj.GetComponent<RoomContents>();
80 if (roomContents != null)

82 allRooms.Add(roomContents);
83 }
84 }
85
86 // 방 번호 할당
87 for (int i = 0; i < allRooms.Count; i++)
88 {
89 if (string.IsNullOrEmpty(allRooms[i].roomID))
90 {
91 allRooms[i].roomID = (i + 101).ToString(); // 101, 102, 103...
92 }
93 }
94
95 DebugLog($"총 {allRooms.Count}개의 방이 감지되었습니다.", true);
96
97 if (allRooms.Count == 0)

168 public void ReportRoomUsage(string aiName, RoomContents room)
169 {
170 if (room == null) return;
171
172 // 방이 이미 사용 중인지 확인
173 if (room.IsRoomUsed)
174 {
175 DebugLog($"{aiName}가 이미 사용 중인 방 {room.roomID}에 접근했습니다.");
176 return;
177 }
178
179 // 방 요금 계산 (방 가격 * 오늘의 배율)
180 int finalPrice = Mathf.RoundToInt(room.UseRoom() * priceMultiplier);
181
182 // 방 명성도 가져오기
183 int roomReputation = room.TotalRoomReputation;
184
185 DebugLog($"방 사용 보고 - AI: {aiName}, 방: {room.roomID}, 가격: {finalPrice}원, 방 명성도: {roomReputation}", true);
186
187 // 로그 추가
188 if (showUsageLogs)
189 {
190 string usageLog = $"{aiName}이(가) 방 {room.roomID}을(를) 사용: {finalPrice}원, 명성도: {roomReputation}";
191 usedRoomLogs.Add(usageLog);
192
193 // 로그 목록 크기 제한 (최근 20개만 유지)
194 if (usedRoomLogs.Count > 20)
195 {
196 usedRoomLogs.RemoveAt(0);
197 }
198 }
199
200 // 결제 시스템에 요금과 명성도 추가
201 if (paymentSystem != null)
202 {
203 DebugLog($"PaymentSystem에 결제 정보 추가 - AI: {aiName}, 방: {room.roomID}, 가격: {finalPrice}, 명성도: {roomReputation}");
204 paymentSystem.AddPayment(aiName, finalPrice, room.roomID, roomReputation);
205 }
206 else
207 {
208 DebugLog("PaymentSystem을 찾을 수 없습니다!", true);
209 }
210 }
