110 Debug.LogError("세이브 데이터 역직렬화 실패");

115 await LoadMainScene();
116 }
117 catch (System.Exception e)
118 {
119 Debug.LogError($"세이브 파일 로드 오류: {e.Message}");
120 await LoadMainScene();

254 private void LoadGridData(GridData gridData, GridDataSave saveData)
255 {
256 if (gridData == null || saveData == null || saveData.placedObjects == null)
257 {
258 Debug.LogError("필수 데이터가 null입니다.");
259 return;
260 }
261
262 gridData.placedObjects = new Dictionary<Vector3Int, List<PlacementData>>();
263 var processedObjects = new HashSet<int>();
264
265 foreach (var entry in saveData.placedObjects)
266 {
267 if (entry.value == null || entry.value.Count == 0)
268 {
269 Debug.LogWarning($"GridEntry의 value가 null 또는 비어 있습니다. key: {entry.key}");
270 continue;
271 }
272
273 var placementData = entry.value[0]; // 첫 번째 PlacementData를 기준으로
274 if (processedObjects.Contains(placementData.PlacedObjectIndex))
275 {
276 continue; // 이미 처리된 객체 무시
277 }
278
279 ObjectData objectData = PlacementSystem.Instance.GetDatabase().GetObjectData(placementData.ID);
280 if (objectData != null)
281 {
282 Vector3 worldPosition = PlacementSystem.Instance.grid.GetCellCenterWorld(entry.key);
283 Debug.Log($"현재 저장된 데이터의 월드 포지션 값 = {worldPosition}");
284
285 // ▼▼▼ [핵심 수정] ▼▼▼
286 // entry.key.y (정수 그리드 좌표)를 사용하여 정확한 층 번호를 계산합니다.
287 int floor = ConvertGridYToFloorNumber(entry.key.y);
288 Debug.Log($"현재 저장된 데이터의 y값 = {floor}와 기존 값 ={entry.key.y}");
289
290 // floor 가 1, 2, 3, 4 (층) 값이 나올때, worldPosition.y 의 값 변화
291 // 1층 0, 2층 4.8175, 3층 9.63405, 4층 14.45
292
293 float floorheight = GetFloorHeight(floor);
294
295 worldPosition.y = floorheight;
296
297 // PlaceObject 호출 시 계산된 층 번호를 세 번째 인자로 전달합니다.
298 int index = ObjectPlacer.Instance.PlaceObject(objectData.Prefab, worldPosition, placementData.Rotation, floor);
299 // ▲▲▲ [핵심 수정] ▲▲▲
300
301 //int index = ObjectPlacer.Instance.PlaceObject(objectData.Prefab, worldPosition, placementData.Rotation);
302
303 if (index != -1)
304 {
305 // ▼ [수정] 새로 생성된 오브젝트의 인덱스를 processedObjects에 추가합니다.
306 processedObjects.Add(index);
307
308 // 원본 PlacementData의 인덱스를 새로 받은 인덱스로 갱신합니다.
309 placementData.PlacedObjectIndex = index;
310
311 // 오브젝트가 점유하는 모든 위치에 올바른 데이터를 다시 추가합니다.
312 List<Vector3Int> occupiedPositions = PlacementSystem.Instance.floorData.CalculatePosition(entry.key, objectData.Size, placementData.Rotation, PlacementSystem.Instance.grid);
313 PlacementData dataToAdd = new PlacementData(occupiedPositions, placementData.ID, index, placementData.KindIndex, placementData.Rotation);
314
315 foreach (var pos in occupiedPositions)
316 {
317 if (!gridData.placedObjects.ContainsKey(pos))
318 {
319 gridData.placedObjects[pos] = new List<PlacementData>();
320 }
321 gridData.placedObjects[pos].Add(dataToAdd);
322 }
323 Debug.Log($"로드 성공 - Index: {index}, ID: {placementData.ID}, Pos: {entry.key}");
324 }
325 else
326 {
327 Debug.LogError($"로드 실패 - key: {entry.key}, ID: {placementData.ID}");
328 }
329 }
330 }
