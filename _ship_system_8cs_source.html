123 private void GenerateShipSchedules()
124 {
125 if (shipRoutes.Count == 0)
126 {
127 DebugLog("설정된 배 루트가 없습니다.", true);
128 return;
129 }
130
131 shipSchedules.Clear();
132
133 // AI 스폰 시간에 맞춰 배 스케줄 생성
134 foreach (var route in shipRoutes)
135 {
136 if (route.IsValid())
137 {
138 // AI 스폰 시간에 맞춰 배 도착 시간 설정
139 float aiSpawnTime = aiSpawner.GetNextSpawnTime();
140 route.arrivalTime = aiSpawnTime;
141
142 var schedule = new ShipSchedule(route);
143 shipSchedules[route.routeId] = schedule;
144 DebugLog($"배 스케줄 생성: {route.routeId} (도착 시간: {aiSpawnTime}분)", showImportantLogsOnly);
145 }
146 }
147
148 DebugLog($"총 {shipSchedules.Count}개의 배 스케줄 생성됨", true);
149 }

154 private void CheckShipSchedules()
155 {
156 if (timeSystem == null) return;
157
158 float currentGameTime = timeSystem.GetCurrentTimeInMinutes();
159
160 foreach (var schedule in shipSchedules.Values)
161 {
162 // 스폰 시간 체크
163 if (!schedule.isShipSpawned && ShouldSpawnShip(schedule, currentGameTime))
164 {
165 SpawnShip(schedule);
166 }
167
168 // 배가 스폰되었지만 아직 활성 상태인지 확인
169 if (schedule.isShipSpawned && schedule.shipController != null)
170 {
171 // GameObject가 파괴되었는지 확인
172 if (schedule.shipController.gameObject == null)
173 {
174 DebugLog($"배 GameObject가 파괴됨 - 스케줄 리셋: {schedule.route.routeId}", true);
175 ResetAndUpdateSchedule(schedule);
176 continue;
177 }
178
179 // 배가 비활성 상태가 되면 (출발 완료) 스케줄 리셋
180 if (schedule.shipController.CurrentState == ShipState.Inactive && schedule.shipController.HasCompletedRoute)
181 {
182 DebugLog($"배 출발 완료 감지 - 풀로 반환: {schedule.route.routeId}", true);
183
184 // 활성 배 목록에서 제거
185 if (activeShips.Contains(schedule.shipController))
186 {
187 activeShips.Remove(schedule.shipController);
188 }
189
190 // 풀로 반환
191 shipPool.ReturnShip(schedule.shipController.gameObject);
192
193 // 스케줄 리셋 및 다음 AI 스폰 시간으로 업데이트
194 ResetAndUpdateSchedule(schedule);
195 }
196 else
197 {
198 // 현재 상태 로그 (디버그용)
199 DebugLog($"배 상태 체크: {schedule.route.routeId} - {schedule.shipController.CurrentState}, 완료: {schedule.shipController.HasCompletedRoute}", false);
200 }
201 }
202 }

217 private void SpawnShip(ShipSchedule schedule)
218 {
219 GameObject shipObj = shipPool.GetShip();
220 if (shipObj == null)
221 {
222 DebugLog("사용 가능한 배가 없습니다.", true);
223 return;
224 }
225
226 ShipController shipController = shipObj.GetComponent<ShipController>();
227 if (shipController == null)
228 {
229 shipController = shipObj.AddComponent<ShipController>();
230 }
231
232 // 배 초기화
233 shipController.Initialize(schedule.route, this);
234 shipController.StartJourney();
235
236 // 스케줄 업데이트
237 schedule.isShipSpawned = true;
238 schedule.shipController = shipController;
239
240 // 활성 배 목록에 추가
241 activeShips.Add(shipController);
242
243 DebugLog($"배 스폰됨: {schedule.route.routeId}", true);
244
245 // 이벤트 발생
246 OnShipSpawned?.Invoke(shipController);
247 }
