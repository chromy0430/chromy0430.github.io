217 private IEnumerator MoveToPosition(Vector3 destination, Quaternion rotation)
218 {
219 Vector3 startPosition = transform.position;
220 Quaternion startRotation = transform.rotation;
221
222 float distance = Vector3.Distance(startPosition, destination);
223 float baseSpeed = assignedRoute.movementSpeed;
224 float journeyTime = distance / baseSpeed;
225
226 float elapsedTime = 0f;
227
228 while (elapsedTime < journeyTime)
229 {
230 elapsedTime += Time.deltaTime;
231 float progress = elapsedTime / journeyTime;
232
233 // 부드러운 이동 곡선 적용
234 float smoothProgress = Mathf.SmoothStep(0f, 1f, progress);
235
236 // 위치 보간
237 Vector3 currentPos = Vector3.Lerp(startPosition, destination, smoothProgress);
238 currentPos.y = 0.5f; // 바닥에서 약간 위로 유지
239 transform.position = currentPos;
240
241 // 회전 보간
242 transform.rotation = Quaternion.Slerp(startRotation, rotation, smoothProgress);
243
244 // 현재 속도 계산 (시각적 표시용)
245 currentSpeed = baseSpeed * (1f - Mathf.Abs(0.5f - smoothProgress) * 0.5f);
246
247 yield return null;
248 }
249
250 // 최종 위치 및 회전 설정
251 destination.y = 0.5f;
252 transform.position = destination;
253 transform.rotation = rotation;
254 currentSpeed = 0f;
255
256 // 웨이포인트 도달 처리
257 OnWaypointReached?.Invoke(this);
258
259 // 현재 웨이포인트 인덱스가 유효한 범위 내인지 확인 후 로그 출력
260 if (currentWaypointIndex >= 0 && currentWaypointIndex < assignedRoute.waypoints.Count && assignedRoute.waypoints[currentWaypointIndex] != null)
261 {
262 DebugLog($"웨이포인트 {currentWaypointIndex} 도달: {assignedRoute.waypoints[currentWaypointIndex].name}", showMovementLogs);
263 }
264 else
265 {
266 DebugLog($"웨이포인트 {currentWaypointIndex} 도달 (정박지로 이동 중)", showMovementLogs);
267 }
268
269 currentWaypointIndex++;
270
271 // 다음 웨이포인트로 이동 또는 정박 시작
272 yield return new WaitForSeconds(0.2f); // 짧은 대기
273 MoveToNextWaypoint();
274 }

307 private IEnumerator WaitForDockingDuration()
308 {
309 if (shipSystem == null)
310 {
311 DebugLog("ShipSystem 참조가 없습니다.", true);
312 yield break;
313 }
314
315 // TimeSystem을 통해 실제 게임 시간으로 대기
316 TimeSystem timeSystem = TimeSystem.Instance;
317 if (timeSystem == null)
318 {
319 DebugLog("TimeSystem을 찾을 수 없습니다.", true);
320 yield break;
321 }
322
323 float dockingDurationMinutes = 30f; // 30분 정박
324 float startTime = (float)timeSystem.GetCurrentTimeInMinutes(); // int를 float로 변환
325 float endTime = startTime + dockingDurationMinutes;
326
327 DebugLog($"정박 대기 시작: {dockingDurationMinutes}분 (게임 시간)", true);
328 DebugLog($"시작 시간: {startTime}분, 종료 시간: {endTime}분, 시간 배속: {timeSystem.timeMultiplier}", true);
329
330 // 게임 시간으로 대기
331 float checkInterval = 1f; // 1초마다 체크
332 float lastCheckTime = Time.time;
333
334 while ((float)timeSystem.GetCurrentTimeInMinutes() < endTime)
335 {
336 // 게임이 일시정지되거나 시간 배속이 0이면 대기
337 if (timeSystem.timeMultiplier <= 0)
338 {
339 DebugLog("시간이 정지되어 있어 대기 중...", false);
340 yield return new WaitForSeconds(0.5f);
341 continue;
342 }
343
344 // 주기적으로 진행 상황 로그
345 if (Time.time - lastCheckTime >= checkInterval)
346 {
347 float currentTime = (float)timeSystem.GetCurrentTimeInMinutes();
348 float remainingTime = endTime - currentTime;
349 DebugLog($"정박 대기 중... 남은 시간: {remainingTime:F1}분", false);
350 lastCheckTime = Time.time;
351 }
352
353 yield return new WaitForSeconds(0.1f);
354 }
355
356 DebugLog("정박 시간 완료 - 출발 준비", true);
357
358 // 자동으로 출발 시작
359 StartDeparture();
360 }

365 private IEnumerator DepartureSequence()
366 {
367 DebugLog("출발 시퀀스 시작", true);
368
369 // 출발 웨이포인트가 있는지 확인
370 if (assignedRoute.departureWaypoints == null || assignedRoute.departureWaypoints.Count == 0)
371 {
372 DebugLog("출발 웨이포인트가 설정되지 않았습니다. 기본 출발 처리를 진행합니다.", true);
373
374 // 출발 웨이포인트가 없으면 현재 위치에서 약간 멀리 이동 후 비활성화
375 Vector3 currentPos = transform.position;
376 Vector3 exitDirection = Vector3.forward; // 기본 출발 방향
377
378 // 도착 경로의 반대 방향으로 설정 (가능한 경우)
379 if (assignedRoute.waypoints.Count >= 2)
380 {
381 Vector3 lastDirection = (assignedRoute.waypoints[assignedRoute.waypoints.Count - 1].position -
382 assignedRoute.waypoints[assignedRoute.waypoints.Count - 2].position).normalized;
383 exitDirection = lastDirection;
384 }
385
386 Vector3 exitPosition = currentPos + exitDirection * 10f; // 10 유닛 멀리
387 exitPosition.y = 0.5f;
388
389 DebugLog($"기본 출발 지점으로 이동: {exitDirection}", true);
390
391 // 출발 지점으로 이동
392 yield return StartCoroutine(MoveToPosition(exitPosition, Quaternion.LookRotation(exitDirection)));
393
394 // 출발 완료
395 currentState = ShipState.Inactive;
396 hasCompletedRoute = true;
397 DebugLog("기본 출발 완료 - 배 비활성화", true);
398 yield break;
399 }
400
401 // 출발 웨이포인트들을 순서대로 이동
402 for (int i = 0; i < assignedRoute.departureWaypoints.Count; i++)
403 {
404 Transform departureWaypoint = assignedRoute.departureWaypoints[i];
405 if (departureWaypoint == null)
406 {
407 DebugLog($"출발 웨이포인트 {i}가 null입니다.", true);
408 continue;
409 }
410
411 Vector3 destination = departureWaypoint.position;
412 Vector3 direction = (destination - transform.position).normalized;
413 Quaternion targetRotation = direction != Vector3.zero ? Quaternion.LookRotation(direction) : transform.rotation;
414
415 DebugLog($"출발 웨이포인트 {i}로 이동: {departureWaypoint.name}", showMovementLogs);
416
417 yield return StartCoroutine(MoveToPosition(destination, targetRotation));
418
419 // 각 웨이포인트 사이에 짧은 대기
420 yield return new WaitForSeconds(0.2f);
421 }
422
423 DebugLog("모든 출발 웨이포인트 통과 완료", true);
424
425 // 출발 완료
426 currentState = ShipState.Inactive;
427 hasCompletedRoute = true;
428
429 DebugLog("출발 완료 - 배 비활성화", true);
430 }
