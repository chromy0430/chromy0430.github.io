67 case QuestCompletionType.EarnMoney:

97 public void OnEndDrag(PointerEventData eventData)
98 {
99 float swipeDistance = eventData.position.x - startDragPosition.x;
100
101 if (swipeDistance > swipeThreshold) // 오른쪽 스와이프 (수락)
102 {
103 QuestManager.Instance.AcceptQuest();
104 }
105 else if (swipeDistance < -swipeThreshold) // 왼쪽 스와이프 (거절)
106 {
107 QuestManager.Instance.DeclineQuest();
108 }
109 else // 원래 위치로 복귀
110 {
111 questPanel.GetComponent<RectTransform>().DOAnchorPos(initialPosition, 0.2f).SetUpdate(true);
112 }
