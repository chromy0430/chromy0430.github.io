442 #region 점유상태 확인
443 private bool CheckPlacementValidity(Vector3Int gridPosition, int selectedObjectIndex, Quaternion rotation)
444 {
445 ObjectData objectToPlace = database.objectsData[selectedObjectIndex];
446 bool placingWall = objectToPlace.IsWall;
447 GameObject prefab = database.objectsData[selectedObjectIndex].Prefab;
448
449 // 1. 소지금 체크
450 int buildPrice = objectToPlace.BuildPrice;
451 if (PlayerWallet.Instance.money < buildPrice)
452 {
453 Debug.Log("배치 불가: 소지금 부족");
454 return false; // 소지금 부족 시 배치 불가능
455 }
456
457 // 2. 배치 영역 확인 (planeBounds) - 기존 로직 유지
458 List<Vector3Int> positionsToCheck = floorData.CalculatePosition(gridPosition, objectToPlace.Size, rotation, grid);
459 foreach (Vector3Int pos in positionsToCheck)
460 {
461 Vector3 worldPos = grid.GetCellCenterWorld(pos);
462 //bool isWithinBounds = planeBounds.Any(bound => bound.Contains(worldPos));
463 bool isWithinBounds = planeBounds.AsValueEnumerable().Any(bound => bound.Contains(worldPos));
464 if (!isWithinBounds)
465 {
466 Debug.Log($"그리드 반경을 벗어남: {pos}");
467 return false;
468 }
469 }
470
471 // 바닥 오브젝트 설치 시
472 if (objectToPlace.kindIndex == 0)
473 {
474 Vector3 worldPosition = grid.GetCellCenterWorld(gridPosition); // 그리드 위치를 월드 좌표로 변환
475 RaycastHit hit;
476
477 Debug.Log("바닥플로어 설치 중");
478
479 // 아래로 짧은 레이캐스트 발사 (0.1f 위에서 시작해 0.2f 거리만큼 검사)
480 if (Physics.Raycast(worldPosition + Vector3.up * 0.1f, Vector3.down, out hit, 0.2f, LayerMask.GetMask("StairCollider")))
481 {
482 if (hit.collider != null)
483 {
484 Debug.Log("설치 불가능: 계단과 충돌");
485 return false; // 계단 콜라이더와 충돌하면 설치 불가
486 }
487 }
488
489 // 바닥 플로어를 배치하려는 경우, 해당 위치에 이미 다른 바닥 플로어가 있는지 확인
490 if (!floorData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
491 {
492 Debug.Log("이미 그 자리에 바닥플로어가 있습니다.");
493 return false;
494 }
495
496 return true;
497 }
498
499 // 장식품 설치 시
500 if (objectToPlace.kindIndex == 3)
501 {
502 // 장식품을 배치하려는 경우, 해당 위치에 장식품이 있는지 확인
503 if (!decoData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
504 {
505 return false;
506 }
507 return true;
508 }
509
510 // 가구를 설치 시
511 if (!placingWall)
512 {
513 // 가구의 실제 월드 좌표 기준 크기와 중심을 계산
514 GameObject tempObject = Instantiate(prefab, grid.GetCellCenterWorld(gridPosition), rotation);
515
516 // 콜라이더를 가져옴
517 Collider objCollider = tempObject.GetComponent<Collider>();
518 if (objCollider is null)
519 {
520 objCollider = tempObject.GetComponentInChildren<Collider>();
521 }
522
523 if (objCollider is null)
524 {
525 Destroy(tempObject);
526 return furnitureData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall);
527 }
528
529 // 충돌 검사를 위해 콜라이더는 유지하고 렌더러는 비활성화
530 Renderer[] renderers = tempObject.GetComponentsInChildren<Renderer>();
531 foreach (Renderer renderer in renderers)
532 {
533 renderer.enabled = false;
534 }
535
536 // 마진 값 조정 (더 작은 값으로 검사 영역 축소)
537 float collisionMargin = 0.05f; // 5cm 마진
538
539 Vector3 colliderCenter = objCollider.bounds.center;
540 Vector3 colliderSize = objCollider.bounds.size - new Vector3(collisionMargin, collisionMargin, collisionMargin);
541
542 // 디버그 시각화 - 실제 검사 영역 확인용
543 Debug.DrawLine(colliderCenter - colliderSize/2, colliderCenter + colliderSize/2, Color.red, 2f);
544
545 // 벽 레이어만 검사
546 int wallLayerMask = LayerMask.GetMask("Wall");
547
548 // 더 정확한 충돌 검사 방법 - 다중 Raycast 사용
549 bool collision = false;
550
551 // 콜라이더 경계 상자의 8개 코너에서 레이캐스트
552 Vector3[] corners = new Vector3[8];
553 corners[0] = new Vector3(-1, -1, -1);
554 corners[1] = new Vector3(1, -1, -1);
555 corners[2] = new Vector3(-1, 1, -1);
556 corners[3] = new Vector3(1, 1, -1);
557 corners[4] = new Vector3(-1, -1, 1);
558 corners[5] = new Vector3(1, -1, 1);
559 corners[6] = new Vector3(-1, 1, 1);
560 corners[7] = new Vector3(1, 1, 1);
561
562 // 콜라이더 경계에서 중심으로 레이캐스트
563 foreach (Vector3 cornerDir in corners)
564 {
565 Vector3 startPos = colliderCenter + Vector3.Scale(cornerDir, colliderSize / 2f);
566 Vector3 direction = -cornerDir.normalized;
567 float distance = colliderSize.magnitude / 2f;
568
569 Debug.DrawRay(startPos, direction * distance, Color.yellow, 2f);
570
571 if (Physics.Raycast(startPos, direction, distance, wallLayerMask))
572 {
573 collision = true;
574 Debug.DrawRay(startPos, direction * distance, Color.red, 2f);
575 break;
576 }
577 }
578
579 // 중심에서 6방향으로 추가 레이캐스트
580 if (!collision)
581 {
582 Vector3[] directions = new Vector3[]
583 {
584 Vector3.right, Vector3.left, Vector3.up,
585 Vector3.down, Vector3.forward, Vector3.back
586 };
587
588 foreach (Vector3 dir in directions)
589 {
590 float distance = Vector3.Scale(colliderSize / 2f, new Vector3(
591 Mathf.Abs(dir.x) > 0.01f ? .7f : 0,
592 Mathf.Abs(dir.y) > 0.01f ? .7f : 0,
593 Mathf.Abs(dir.z) > 0.01f ? .7f : 0
594 )).magnitude;
595
596 Debug.DrawRay(colliderCenter, dir * distance, Color.yellow, 2f);
597
598 if (Physics.Raycast(colliderCenter, dir, distance, wallLayerMask))
599 {
600 collision = true;
601 Debug.DrawRay(colliderCenter, dir * distance, Color.red, 2f);
602 break;
603 }
604 }
605 }
606
607 // 충돌이 감지되면 배치 불가
608 if (collision)
609 {
610 Debug.Log($"배치 불가: 가구가 벽과 충돌했습니다. 위치: {gridPosition}");
611 Destroy(tempObject);
612 return false;
613 }
614
615 // 오버랩박스로 마지막 확인 (더 작은 크기로)
616 Collider[] hitColliders = Physics.OverlapBox(
617 colliderCenter,
618 colliderSize / 4f, // 더 작은 영역으로 검사
619 rotation,
620 wallLayerMask
621 );
622
623 // 충돌 감지 시 로그 출력 및 배치 금지
624 if (hitColliders.Length > 0)
625 {
626 //Debug.Log($"배치 불가: 가구가 벽과 충돌했습니다. 벽 개수: {hitColliders.Length}");
627 foreach (Collider hit in hitColliders)
628 {
629 Debug.Log($"- 충돌한 벽 오브젝트: {hit.gameObject.name}");
630 }
631 Destroy(tempObject);
632 return false;
633 }
634
635 // 임시 오브젝트 제거
636 Destroy(tempObject);
637
638 // GridData 기반 충돌 체크 - 다른 가구와의 충돌 확인
639 if (!furnitureData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
640 {
641 Debug.Log($"배치 불가: 해당 위치에 가구가 이미 존재합니다. 위치: {gridPosition}");
642 return false;
643 }
644 }
645
646 if (placingWall)
647 {
648 // 1. 기존 다른 벽과의 충돌 검사
649 if (!wallData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
650 {
651 Debug.Log($"벽 배치 불가: 같은 각도의 벽 충돌 at {gridPosition}");
652 return false;
653 }
654 Debug.Log($"벽-벽 충돌 검사 통과 at {gridPosition}");
655
656 // 2. 벽과 가구 간 충돌 검사 (Raycast 기반)
657 GameObject tempObject = Instantiate(prefab, grid.GetCellCenterWorld(gridPosition), rotation);
658
659 Collider wallCollider = tempObject.GetComponent<Collider>();
660 if (wallCollider is null)
661 {
662 wallCollider = tempObject.GetComponentInChildren<Collider>();
663 }
664
665 if (wallCollider is null)
666 {
667 Debug.Log($"벽 배치 불가: 콜라이더 없음 at {gridPosition}");
668 Destroy(tempObject);
669 return false;
670 }
671 Debug.Log($"콜라이더 확인 완료: {wallCollider.gameObject.name}");
672
673 // 충돌 검사를 위해 렌더러 비활성화
674 Renderer[] renderers = tempObject.GetComponentsInChildren<Renderer>();
675 foreach (Renderer renderer in renderers)
676 {
677 renderer.enabled = false;
678 }
679
680
681
682 // 벽 콜라이더의 중심과 크기
683 Vector3 colliderCenter = wallCollider.bounds.center;
684 Vector3 colliderSize = wallCollider.bounds.size;
685
686 // 벽의 바닥 위치 계산 (Raycast 발사 위치)
687 Vector3 wallBottomCenter = colliderCenter - tempObject.transform.up * (colliderSize.y / 5f);
688
689 // 디버그 시각화: 콜라이더 범위 및 발사 위치 표시
690 Debug.DrawLine(colliderCenter - colliderSize / 2, colliderCenter + colliderSize / 2, Color.blue, 2f);
691 Debug.DrawLine(wallBottomCenter - new Vector3(0.1f, 0, 0.1f), wallBottomCenter + new Vector3(0.1f, 0, 0.1f), Color.green, 2f);
692
693 Vector3 leftRaycastOrigin1 = wallBottomCenter - tempObject.transform.forward * 1;
694 Vector3 rightRaycastOrigin2 = wallBottomCenter + tempObject.transform.forward * 1;
695
696 // Raycast 방향 설정 (벽의 로컬 아래 방향)
697 Vector3 raycastDirection = -tempObject.transform.up;
698
699 // Raycast 거리 계산: 벽 바닥에서 그리드 바닥까지의 거리
700 float distance = Mathf.Abs(wallBottomCenter.y - grid.GetCellCenterWorld(gridPosition).y) + 1f;
701
702 // 각 위치에서 Raycast 발사
703 Vector3[] raycastOrigins = new Vector3[]
704 {
705 leftRaycastOrigin1, rightRaycastOrigin2
706 };
707
708 bool collision = false;
709 foreach (Vector3 origin in raycastOrigins)
710 {
711 Debug.DrawRay(origin, raycastDirection * distance, Color.yellow, 2f);
712 if (Physics.Raycast(origin, raycastDirection, distance, furnitureLayerMask))
713 {
714 collision = true;
715 Debug.DrawRay(origin, raycastDirection * distance, Color.red, 2f);
716 Debug.Log($"벽 배치 불가: Raycast로 가구와 충돌 감지됨 at {gridPosition}, 방향: {raycastDirection}, 발사 위치: {origin}");
717 break;
718 }
719 }
720
721 if (!collision)
722 {
723 Debug.Log($"벽-가구 충돌 검사 통과 at {gridPosition}");
724 }
725
726 // 충돌이 감지되면 벽 설치 차단
727 if (collision)
728 {
729 Destroy(tempObject);
730 return false;
731 }
732
733 Destroy(tempObject);
734 Debug.Log($"벽 배치 가능: 모든 검사 통과 at {gridPosition}");
735 }
736
737 Debug.Log("벽 검사 끝");
738
739 // 모든 검사를 통과하면 배치 가능
740 return true;
741 }

1083 for (int i = 0; i < stepCount; i++)
1084 {

1096 currentPos += stepDirection;

1102 }
