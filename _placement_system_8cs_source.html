395 #region 점유상태 확인
396 private bool CheckPlacementValidity(Vector3Int gridPosition, int selectedObjectIndex, Quaternion rotation)
397 {
398 ObjectData objectToPlace = database.objectsData[selectedObjectIndex];
399 bool placingWall = objectToPlace.IsWall;
400 GameObject prefab = database.objectsData[selectedObjectIndex].Prefab;
401
402 // 1. 배치 영역 확인 (planeBounds) - 기존 로직 유지
403 List<Vector3Int> positionsToCheck = floorData.CalculatePosition(gridPosition, objectToPlace.Size, rotation, grid);
404 foreach (Vector3Int pos in positionsToCheck)
405 {
406 Vector3 worldPos = grid.GetCellCenterWorld(pos);
407 //bool isWithinBounds = planeBounds.Any(bound => bound.Contains(worldPos));
408 bool isWithinBounds = planeBounds.AsValueEnumerable().Any(bound => bound.Contains(worldPos));
409 if (!isWithinBounds)
410 {
411 //Debug.Log($"그리드 반경을 벗어남: {pos}");
412 return false;
413 }
414 }
415
416 if (objectToPlace.kindIndex == 0)
417 {
418 // 바닥 플로어를 배치하려는 경우, 해당 위치에 이미 다른 바닥 플로어가 있는지 확인
419 if (!floorData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
420 {
421 //Debug.Log($"배치 불가: 해당 위치에 이미 바닥 플로어가 존재합니다. 위치: {gridPosition}");
422 return false;
423 }
424 return true;
425 }
426
427 if (objectToPlace.kindIndex == 3)
428 {
429 // 바닥 플로어를 배치하려는 경우, 해당 위치에 이미 다른 바닥 플로어가 있는지 확인
430 if (!decoData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
431 {
432 //Debug.Log($"배치 불가: 해당 위치에 이미 바닥 플로어가 존재합니다. 위치: {gridPosition}");
433 return false;
434 }
435 return true;
436 }
437
438 if (!placingWall) // 가구를 설치하려는 경우
439 {
440 // 가구의 실제 월드 좌표 기준 크기와 중심을 계산
441
442 GameObject tempObject = Instantiate(prefab, grid.GetCellCenterWorld(gridPosition), rotation);
443
444 // 콜라이더를 가져옴
445 Collider objCollider = tempObject.GetComponent<Collider>();
446 if (objCollider is null)
447 {
448 objCollider = tempObject.GetComponentInChildren<Collider>();
449 }
450
451 if (objCollider is null)
452 {
453 //Debug.LogWarning($"오브젝트 {prefab.name}에 콜라이더가 없습니다. 충돌 검사를 건너뜁니다.");
454 Destroy(tempObject);
455 return furnitureData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall);
456 }
457
458 // 충돌 검사를 위해 콜라이더는 유지하고 렌더러는 비활성화
459 Renderer[] renderers = tempObject.GetComponentsInChildren<Renderer>();
460 foreach (Renderer renderer in renderers)
461 {
462 renderer.enabled = false;
463 }
464
465 // 마진 값 조정 (더 작은 값으로 검사 영역 축소)
466 float collisionMargin = 0.05f; // 5cm 마진
467
468 Vector3 colliderCenter = objCollider.bounds.center;
469 Vector3 colliderSize = objCollider.bounds.size - new Vector3(collisionMargin, collisionMargin, collisionMargin);
470
471 // 디버그 시각화 - 실제 검사 영역 확인용
472 Debug.DrawLine(colliderCenter - colliderSize/2, colliderCenter + colliderSize/2, Color.red, 2f);
473
474 // 벽 레이어만 검사
475 int wallLayerMask = LayerMask.GetMask("Wall");
476
477 // 더 정확한 충돌 검사 방법 - 다중 Raycast 사용
478 bool collision = false;
479
480 // 콜라이더 경계 상자의 8개 코너에서 레이캐스트
481 Vector3[] corners = new Vector3[8];
482 corners[0] = new Vector3(-1, -1, -1);
483 corners[1] = new Vector3(1, -1, -1);
484 corners[2] = new Vector3(-1, 1, -1);
485 corners[3] = new Vector3(1, 1, -1);
486 corners[4] = new Vector3(-1, -1, 1);
487 corners[5] = new Vector3(1, -1, 1);
488 corners[6] = new Vector3(-1, 1, 1);
489 corners[7] = new Vector3(1, 1, 1);
490
491 // 콜라이더 경계에서 중심으로 레이캐스트
492 foreach (Vector3 cornerDir in corners)
493 {
494 Vector3 startPos = colliderCenter + Vector3.Scale(cornerDir, colliderSize / 2f);
495 Vector3 direction = -cornerDir.normalized;
496 float distance = colliderSize.magnitude / 2f;
497 Debug.Log($"{distance} 레이 거리");
498
499 Debug.DrawRay(startPos, direction * distance, Color.yellow, 2f);
500
501 if (Physics.Raycast(startPos, direction, distance, wallLayerMask))
502 {
503 collision = true;
504 Debug.DrawRay(startPos, direction * distance, Color.red, 2f);
505 break;
506 }
507 }
508
509 // 중심에서 6방향으로 추가 레이캐스트
510 if (!collision)
511 {
512 Vector3[] directions = new Vector3[]
513 {
514 Vector3.right, Vector3.left, Vector3.up,
515 Vector3.down, Vector3.forward, Vector3.back
516 };
517
518 foreach (Vector3 dir in directions)
519 {
520 float distance = Vector3.Scale(colliderSize / 2f, new Vector3(
521 Mathf.Abs(dir.x) > 0.01f ? .7f : 0,
522 Mathf.Abs(dir.y) > 0.01f ? .7f : 0,
523 Mathf.Abs(dir.z) > 0.01f ? .7f : 0
524 )).magnitude;
525
526 Debug.DrawRay(colliderCenter, dir * distance, Color.yellow, 2f);
527
528 if (Physics.Raycast(colliderCenter, dir, distance, wallLayerMask))
529 {
530 collision = true;
531 Debug.DrawRay(colliderCenter, dir * distance, Color.red, 2f);
532 break;
533 }
534 }
535 }
536
537 // 충돌이 감지되면 배치 불가
538 if (collision)
539 {
540 Debug.Log($"배치 불가: 가구가 벽과 충돌했습니다. 위치: {gridPosition}");
541 Destroy(tempObject);
542 return false;
543 }
544
545 // 오버랩박스로 마지막 확인 (더 작은 크기로)
546 Collider[] hitColliders = Physics.OverlapBox(
547 colliderCenter,
548 colliderSize / 4f, // 더 작은 영역으로 검사
549 rotation,
550 wallLayerMask
551 );
552
553 // 충돌 감지 시 로그 출력 및 배치 금지
554 if (hitColliders.Length > 0)
555 {
556 //Debug.Log($"배치 불가: 가구가 벽과 충돌했습니다. 벽 개수: {hitColliders.Length}");
557 foreach (Collider hit in hitColliders)
558 {
559 Debug.Log($"- 충돌한 벽 오브젝트: {hit.gameObject.name}");
560 }
561 Destroy(tempObject);
562 return false;
563 }
564
565 // 임시 오브젝트 제거
566 Destroy(tempObject);
567
568 // GridData 기반 충돌 체크 - 다른 가구와의 충돌 확인
569 if (!furnitureData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
570 {
571 //Debug.Log($"배치 불가: 해당 위치에 가구가 이미 존재합니다. 위치: {gridPosition}");
572 return false;
573 }
574 }
575
576 if (placingWall)
577 {
578 // 1. 기존 다른 벽과의 충돌 검사
579 if (!wallData.CanPlaceObjectAt(gridPosition, objectToPlace.Size, rotation, grid, placingWall))
580 {
581 Debug.Log($"벽 배치 불가: 같은 각도의 벽 충돌 at {gridPosition}");
582 return false;
583 }
584
585 // 2. 벽과 가구 간 충돌 검사 (Raycast 기반)
586 GameObject tempObject = Instantiate(prefab, grid.GetCellCenterWorld(gridPosition), rotation);
587
588 Collider wallCollider = tempObject.GetComponent<Collider>();
589 if (wallCollider is null)
590 {
591 wallCollider = tempObject.GetComponentInChildren<Collider>();
592 }
593
594 if (wallCollider is not null)
595 {
596 // 충돌 검사를 위해 렌더러 비활성화
597 Renderer[] renderers = tempObject.GetComponentsInChildren<Renderer>();
598 foreach (Renderer renderer in renderers)
599 {
600 renderer.enabled = false;
601 }
602
603 // 가구 레이어 마스크 설정
604 int furnitureLayerMask = LayerMask.GetMask("Furniture");
605
606 // 벽 콜라이더의 중심과 크기
607 Vector3 colliderCenter = wallCollider.bounds.center;
608 Vector3 colliderSize = wallCollider.bounds.size;
609
610 // 벽의 바닥 위치 계산 (Raycast 발사 위치)
611 Vector3 wallBottomCenter = colliderCenter - tempObject.transform.up * (colliderSize.y / 5f);
612
613 // 디버그 시각화: 콜라이더 범위 및 발사 위치 표시
614 Debug.DrawLine(colliderCenter - colliderSize/2, colliderCenter + colliderSize/2, Color.blue, 2f); // 콜라이더 범위
615 Debug.DrawLine(wallBottomCenter - new Vector3(0.1f, 0, 0.1f), wallBottomCenter + new Vector3(0.1f, 0, 0.1f), Color.green, 2f); // 발사 위치
616
617
618 Vector3 leftRaycastOrigin1 = wallBottomCenter - tempObject.transform.forward * 1; // 왼쪽 첫 번째 위치
619 Vector3 rightRaycastOrigin2 = wallBottomCenter + tempObject.transform.forward * 1; // 오른쪽 두 번째 위치
620
621 // Raycast 방향 설정 (벽의 로컬 아래 방향)
622 Vector3 raycastDirection = -tempObject.transform.up;
623
624 // Raycast 거리 계산: 벽 바닥에서 그리드 바닥까지의 거리
625 float distance = Mathf.Abs(wallBottomCenter.y - grid.GetCellCenterWorld(gridPosition).y) + 1f; // 바닥까지 거리 + 여유분
626
627 // 각 위치에서 Raycast 발사
628 Vector3[] raycastOrigins = new Vector3[]
629 {
630 leftRaycastOrigin1, rightRaycastOrigin2
631 };
632
633 bool collision = false;
634 foreach (Vector3 origin in raycastOrigins)
635 {
636 Debug.DrawRay(origin, raycastDirection * distance, Color.yellow, 2f); // 디버그: 노란색으로 Raycast 경로 표시
637
638 if (Physics.Raycast(origin, raycastDirection, distance, furnitureLayerMask))
639 {
640 collision = true;
641 Debug.DrawRay(origin, raycastDirection * distance, Color.red, 2f); // 디버그: 충돌 시 빨간색으로 표시
