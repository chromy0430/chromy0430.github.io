122 // 태그된 오브젝트들이 변경되었는지 확인하는 메서드
123 bool HasTaggedObjectsChanged()
124 {
125 foreach (var tag in tagsToBake)
126 {
127 if (string.IsNullOrEmpty(tag)) continue;
128
129 var currentObjects = GameObject.FindGameObjectsWithTag(tag);
130
131 if (!tagObjectCache.ContainsKey(tag) ||
132 tagObjectCache[tag].Count != currentObjects.Length)
133 {
134 return true;
135 }
136
137 // 실제 오브젝트들이 같은지 확인
138 var cachedObjects = tagObjectCache[tag];
139 for (int i = 0; i < currentObjects.Length; i++)
140 {
141 if (!cachedObjects.Contains(currentObjects[i]))
142 {
143 return true;
144 }
145 }
146 }
147
148 return false;
