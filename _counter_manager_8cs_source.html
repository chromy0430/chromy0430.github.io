64 private void RemoveFromQueue(Queue<AIAgent> queue, AIAgent agent)
65 {
66 int originalCount = queue.Count;
67 var tempQueue = new Queue<AIAgent>();
68 bool removed = false;
69
70 while (queue.Count > 0)
71 {
72 var queuedAgent = queue.Dequeue();
73 if (queuedAgent != agent)
74 {
75 tempQueue.Enqueue(queuedAgent);
76 }
77 else
78 {
79 removed = true;
80 Debug.Log($"[CounterManager] 대기열에서 AI {agent.gameObject.name} 제거됨");
81 }
82 }
83 while (tempQueue.Count > 0)
84 {
85 queue.Enqueue(tempQueue.Dequeue());
86 }
87
88 if (!removed)
89 {
90 Debug.LogWarning($"[CounterManager] AI {agent.gameObject.name}이(가) 대기열에 없어서 제거할 수 없습니다.");
91 }

191 int originalCount = waitingQueue.Count;

194 while (waitingQueue.Count > 0)
