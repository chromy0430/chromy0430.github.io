56 private void RemoveFromQueue(Queue<AIAgent> queue, AIAgent agent)
57 {
58 var tempQueue = new Queue<AIAgent>();
59 while (queue.Count > 0)
60 {
61 var queuedAgent = queue.Dequeue();
62 if (queuedAgent != agent)
63 {
64 tempQueue.Enqueue(queuedAgent);
65 }
66 }
67 while (tempQueue.Count > 0)
68 {
69 queue.Enqueue(tempQueue.Dequeue());
70 }
