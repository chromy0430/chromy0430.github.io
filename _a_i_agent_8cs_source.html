230 {
231 if (rooms != null && rooms.Length > 0)
232 {
233 UpdateRoomList(rooms);
234 }
235 };
236 }
237 Debug.Log($"AI {gameObject.name}: RoomDetector로 룸 감지 시작.");
238

240 }
241 else
242 {
243 // 태그로 룸 찾기
244 GameObject[] taggedRooms = GameObject.FindGameObjectsWithTag("Room");
245 foreach (GameObject room in taggedRooms)
246 {
247 if (!roomList.Any(r => r.gameObject == room))
248 {
249 roomList.Add(new RoomInfo(room));
250 }
251 }
252 Debug.Log($"AI {gameObject.name}: 태그로 {roomList.Count}개 룸 발견.");
253 }
254
255 if (roomList.Count == 0)

265 public static void UpdateRoomList(GameObject[] newRooms)
266 {
267 if (newRooms == null || newRooms.Length == 0) return;
268
269 lock (lockObject)
270 {
271 bool isUpdated = false;
272 HashSet<string> processedRoomIds = new HashSet<string>();
273 List<RoomInfo> updatedRoomList = new List<RoomInfo>();
274
275 foreach (GameObject room in newRooms)
276 {
277 if (room != null)
278 {
279 RoomInfo newRoom = new RoomInfo(room);
280 if (!processedRoomIds.Contains(newRoom.roomId))
281 {
282 processedRoomIds.Add(newRoom.roomId);
283 var existingRoom = roomList.FirstOrDefault(r => r.roomId == newRoom.roomId);
284 if (existingRoom != null)
285 {
286 newRoom.isOccupied = existingRoom.isOccupied;
287 updatedRoomList.Add(newRoom);
288 Debug.Log($"룸 업데이트: {newRoom.roomId}");
289 }
290 else
291 {
292 updatedRoomList.Add(newRoom);
293 isUpdated = true;
294 Debug.Log($"새 룸 추가: {newRoom.roomId}");
295 }
296 }
297 }
298 }
299
300 if (updatedRoomList.Count > 0)
301 {
302 roomList = updatedRoomList;
303 Debug.Log($"룸 리스트 업데이트 완료. 총 룸 수: {roomList.Count}");
304 foreach (var room in roomList)
305 {
306 Debug.Log($"- 룸 ID: {room.roomId}, 사용 중: {room.isOccupied}");
307 }
308 }
309 }

319 void Update()
320 {
321 if (!agent.isOnNavMesh)
322 {
323 Debug.LogWarning($"AI {gameObject.name}: NavMesh 벗어남");
324 ReturnToPool();
325 return;
326 }
327
328 switch (currentState)
329 {
330 case AIState.Wandering:
331 // 배회는 코루틴에서 처리
332 break;
333 case AIState.MovingToQueue:
334 case AIState.WaitingInQueue:
335 // 대기열 동작은 코루틴에서 처리
336 break;
337 case AIState.MovingToRoom:
338 if (!agent.pathPending && agent.remainingDistance < arrivalDistance)
339 {
340 StartCoroutine(UseRoom());
341 }
342 break;
343 case AIState.UsingRoom:
344 // 방 사용은 코루틴에서 처리
345 break;
346 case AIState.ReportingRoom:
347 // 보고는 코루틴에서 처리
348 break;
349 case AIState.ReturningToSpawn:
350 if (!agent.pathPending && agent.remainingDistance < arrivalDistance)
351 {
352 Debug.Log($"AI {gameObject.name}: 스폰 지점 도착, 풀로 반환.");
353 ReturnToPool();
354 }
355 break;
356 }

360 #region 대기열 동작
361 private IEnumerator QueueBehavior()
362 {
363 // CounterManager 또는 카운터가 없으면 배회 또는 디스폰
364 if (counterManager == null || counterPosition == null)
365 {
366 float randomValue = Random.value;
367 if (randomValue < 0.5f)
368 {
369 TransitionToState(AIState.Wandering);
370 wanderingCoroutine = StartCoroutine(WanderingBehavior());
371 Debug.Log($"AI {gameObject.name}: CounterManager/카운터 없음, 배회로 전환.");
372 }
373 else
374 {
375 TransitionToState(AIState.ReturningToSpawn);
376 agent.SetDestination(spawnPoint.position);
377 Debug.Log($"AI {gameObject.name}: CounterManager/카운터 없음, 스폰 지점으로 복귀.");
378 }
379 yield break;
380 }
381
382 // 기존 대기열 로직
383 if (!counterManager.TryJoinQueue(this))
384 {
385 if (currentRoomIndex == -1)
386 {
387 if (Random.value < 0.5f)
388 {
389 TransitionToState(AIState.Wandering);
390 wanderingCoroutine = StartCoroutine(WanderingBehavior());
391 }
392 else
393 {
394 TransitionToState(AIState.ReturningToSpawn);
395 agent.SetDestination(spawnPoint.position);
396 }
397 }
398 else
399 {
400 yield return new WaitForSeconds(Random.Range(1f, 3f));
401 StartCoroutine(QueueBehavior());
402 }
403 yield break;
404 }
405
406 isInQueue = true;
407 TransitionToState(AIState.WaitingInQueue);
408
409 while (isInQueue)
410 {
411 if (!agent.pathPending && agent.remainingDistance < arrivalDistance)
412 {
413 if (counterManager.CanReceiveService(this))
414 {
415 counterManager.StartService(this);
416 isWaitingForService = true;
417
418 while (isWaitingForService)
419 {
420 yield return new WaitForSeconds(0.1f);
421 }
422
423 if (currentRoomIndex != -1)
424 {
425 roomList[currentRoomIndex].isOccupied = false;
426 currentRoomIndex = -1;
427 TransitionToState(AIState.ReturningToSpawn);
428 agent.SetDestination(spawnPoint.position);
429 }
430 else
431 {
432 if (TryAssignRoom())
433 {
434 TransitionToState(AIState.MovingToRoom);
435 agent.SetDestination(roomList[currentRoomIndex].transform.position);
436 }
437 else
438 {
439 if (Random.value < 0.5f)
440 {
441 TransitionToState(AIState.Wandering);
442 wanderingCoroutine = StartCoroutine(WanderingBehavior());
443 }
444 else
445 {
446 TransitionToState(AIState.ReturningToSpawn);
447 agent.SetDestination(spawnPoint.position);
448 }
449 }
450 }
451 break;
452 }
453 }
454 yield return new WaitForSeconds(0.1f);
455 }
456 }

458 private bool TryAssignRoom()
459 {
460 lock (lockObject)
461 {
462 var availableRooms = roomList.Select((room, index) => new { room, index })
463 .Where(r => !r.room.isOccupied)
464 .Select(r => r.index)
465 .ToList();
466
467 if (availableRooms.Count == 0)
468 {
469 Debug.Log($"AI {gameObject.name}: 사용 가능한 룸 없음.");
470 return false;
471 }
472
473 int selectedRoomIndex = availableRooms[Random.Range(0, availableRooms.Count)];
474 if (!roomList[selectedRoomIndex].isOccupied)
475 {
476 roomList[selectedRoomIndex].isOccupied = true;
477 currentRoomIndex = selectedRoomIndex;
478 Debug.Log($"AI {gameObject.name}: 룸 {selectedRoomIndex + 1}번 배정됨.");
479 return true;
480 }
481
482 Debug.Log($"AI {gameObject.name}: 룸 {selectedRoomIndex + 1}번 이미 사용 중.");
483 return false;
484 }

489 private void TransitionToState(AIState newState)
490 {
491 // 기존 코루틴 정리
492 CleanupCoroutines();
493 if (currentState == AIState.UsingRoom)
494 {
495 isBeingServed = false;
496 }
497
498 currentState = newState;
499 currentDestination = GetStateDescription(newState);
500 Debug.Log($"AI {gameObject.name}: 상태 변경: {newState}");
501
502 switch (newState)
503 {
504 case AIState.Wandering:
505 wanderingCoroutine = StartCoroutine(WanderingBehavior());
506 break;
507 case AIState.MovingToRoom:
508 if (currentRoomIndex != -1)
509 {
510 agent.SetDestination(roomList[currentRoomIndex].transform.position);
511 }
512 break;
513 case AIState.ReturningToSpawn:
514 agent.SetDestination(spawnPoint.position);
515 break;
516 }
517 }

535 #region 룸 사용
536 private IEnumerator UseRoom()
537 {
538 float roomUseTime = Random.Range(25f, 35f);
539 float elapsedTime = 0f;
540 bool stayInRoom = Random.value < 0.5f;
541 float wanderRadius = 5f;
542
543 if (currentRoomIndex < 0 || currentRoomIndex >= roomList.Count)
544 {
545 Debug.LogError($"AI {gameObject.name}: 잘못된 룸 인덱스 {currentRoomIndex}.");
546 StartCoroutine(ReportRoomVacancy());
547 yield break;
548 }
549
550 var roomManager = FindObjectOfType<RoomManager>();
551 var room = roomList[currentRoomIndex].gameObject.GetComponent<RoomContents>();
552 if (roomManager != null && room != null)
553 {
554 roomManager.ReportRoomUsage(gameObject.name, room);
555 }
556
557 TransitionToState(AIState.UsingRoom);
558
559 if (stayInRoom)
560 {
561 Debug.Log($"AI {gameObject.name}: 룸 {currentRoomIndex + 1}번 내부 배회.");
562 while (elapsedTime < roomUseTime && agent.isOnNavMesh)
563 {
564 Vector3 roomCenter = roomList[currentRoomIndex].transform.position;
565 float roomSize = roomList[currentRoomIndex].size;
566 if (TryGetValidPosition(roomCenter, roomSize, NavMesh.AllAreas, out Vector3 targetPos))
567 {
568 agent.SetDestination(targetPos);
569 }
570 else
571 {
572 Debug.LogWarning($"AI {gameObject.name}: 룸 {currentRoomIndex + 1}번 내부 유효 위치 없음. 외부 배회로 전환.");
573 stayInRoom = false;
574 break;
575 }
576
577 yield return new WaitForSeconds(Random.Range(2f, 5f));
578 elapsedTime += Random.Range(2f, 5f);
579 }
580 }
581
582 if (!stayInRoom)
583 {
584 Debug.Log($"AI {gameObject.name}: 룸 {currentRoomIndex + 1}번 외부 배회.");
585 while (elapsedTime < roomUseTime && agent.isOnNavMesh)
586 {
587 Vector3 roomCenter = roomList[currentRoomIndex].transform.position;
588 Vector3 randomPoint = roomCenter + Random.insideUnitSphere * wanderRadius;
589 randomPoint.y = transform.position.y;
590
591 if (NavMesh.SamplePosition(randomPoint, out NavMeshHit hit, wanderRadius, NavMesh.AllAreas))
592 {
593 agent.SetDestination(hit.position);
594 }
595
596 yield return new WaitForSeconds(Random.Range(3f, 7f));
597 elapsedTime += Random.Range(3f, 7f);
598 }
599 }
600
601 Debug.Log($"AI {gameObject.name}: 룸 {currentRoomIndex + 1}번 사용 완료, 보고 시작.");
602 StartCoroutine(ReportRoomVacancy());
603 }

605 private IEnumerator ReportRoomVacancy()
606 {
607 TransitionToState(AIState.ReportingRoom);
608 int reportingRoomIndex = currentRoomIndex;
609 Debug.Log($"AI {gameObject.name}: 룸 {reportingRoomIndex + 1}번 사용 완료 보고.");
610
611 lock (lockObject)
612 {
613 if (reportingRoomIndex >= 0 && reportingRoomIndex < roomList.Count)
614 {
615 roomList[reportingRoomIndex].isOccupied = false;
616 currentRoomIndex = -1;
617 Debug.Log($"룸 {reportingRoomIndex + 1}번 비워짐.");
618 }
619 }
620
621 var roomManager = FindObjectOfType<RoomManager>();
622 if (roomManager != null)
623 {
624 int amount = roomManager.ProcessRoomPayment(gameObject.name);
625 Debug.Log($"AI {gameObject.name}: 룸 결제 완료, 금액: {amount}원");
626 }
627
628 StartCoroutine(QueueBehavior());
629
630 var paymentSystem = FindObjectOfType<PaymentSystem>();
631 if (paymentSystem != null && !paymentSystem.HasUnpaidPayments(gameObject.name))
632 {
633 Debug.Log($"AI {gameObject.name}: 모든 결제 완료.");
634 }
635 else
636 {
637 Debug.LogWarning($"AI {gameObject.name}: 미결제 금액 존재.");
638 }
639
640 yield break;
641 }

645 private void WanderOnGround()
646 {
647 Vector3 randomPoint = transform.position + Random.insideUnitSphere * 10f;
648 int groundMask = NavMesh.GetAreaFromName("Ground");
649 if (groundMask == 0)
650 {
651 Debug.LogError($"AI {gameObject.name}: Ground NavMesh 영역 설정되지 않음. 배회 중단.");
652 return;
653 }
654
655 if (NavMesh.SamplePosition(randomPoint, out NavMeshHit hit, 10f, groundMask))
656 {
657 agent.SetDestination(hit.position);
658 Debug.Log($"AI {gameObject.name}: 새로운 배회 위치로 이동.");
659 }

662 private IEnumerator WanderingBehavior()
663 {
664 float wanderingTime = Random.Range(15f, 30f);
665 float elapsedTime = 0f;
666
667 while (currentState == AIState.Wandering && elapsedTime < wanderingTime)
668 {

670 float waitTime = Random.Range(3f, 7f);

682 private bool TryGetValidPosition(Vector3 center, float radius, int layerMask, out Vector3 result)
683 {
684 result = center;
685 float searchRadius = radius * 0.8f;
686
687 for (int i = 0; i < maxRetries; i++)
688 {
689 Vector2 randomCircle = Random.insideUnitCircle * searchRadius;

692 if (NavMesh.SamplePosition(randomPoint, out NavMeshHit hit, searchRadius, layerMask))
693 {
694 if (Vector3.Distance(hit.position, center) <= searchRadius)
695 {
696 result = hit.position;
697 return true;
698 }
699 }
700 }
701 return false;
