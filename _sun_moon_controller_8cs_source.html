152 private void UpdateCelestialBodies(int hour, int minute)
153 {
154 // 정확한 시간 계산 (분 단위 포함)
155 float currentTime = hour + (minute / 60f);
156
157 // 해의 각도 계산 - 수정됨
158 // 12시(정오)에 90도(가장 위), 0시(자정)에 -90도(가장 아래)
159 // 6시: 0도(수평), 18시: 180도(수평 반대편)
160 targetSunAngle = (currentTime * 15f) - 90f; // 15도/시간 (360도/24시간)
161
162 // 달의 각도 계산 (해와 정확히 반대편)
163 targetMoonAngle = targetSunAngle + 180f;
164
165 // 각도 정규화 (-180 ~ 180)
166 targetSunAngle = NormalizeAngle(targetSunAngle);
167 targetMoonAngle = NormalizeAngle(targetMoonAngle);
168
169 // 즉시 적용 또는 부드러운 회전 설정
170 if (!smoothRotation)
171 {
172 currentSunAngle = targetSunAngle;
173 currentMoonAngle = targetMoonAngle;
174 ApplyRotation();
175 }
176
177 // 디버그 정보 출력
178 if (showDebugInfo)
179 {
180 Debug.Log($"시간: {hour:00}:{minute:00} | 해 각도: {currentSunAngle:F1}°/{targetSunAngle:F1}° | 달 각도: {currentMoonAngle:F1}°/{targetMoonAngle:F1}°");
181 }
182 }

187 private void UpdateSmoothRotation()
188 {
189 // 해의 부드러운 회전
190 float sunAngleDiff = Mathf.DeltaAngle(currentSunAngle, targetSunAngle);
191 if (Mathf.Abs(sunAngleDiff) > 0.1f)
192 {
193 currentSunAngle += sunAngleDiff * rotationSpeed * Time.deltaTime;
194 // 각도 정규화 추가
195 currentSunAngle = NormalizeAngle(currentSunAngle);
196 }
197 else
198 {
199 currentSunAngle = targetSunAngle;
200 }
201
202 // 달의 부드러운 회전
203 float moonAngleDiff = Mathf.DeltaAngle(currentMoonAngle, targetMoonAngle);
204 if (Mathf.Abs(moonAngleDiff) > 0.1f)
205 {
206 currentMoonAngle += moonAngleDiff * rotationSpeed * Time.deltaTime;
207 // 각도 정규화 추가
208 currentMoonAngle = NormalizeAngle(currentMoonAngle);
209 }
210 else
211 {
212 currentMoonAngle = targetMoonAngle;
213 }
214
215 // 회전 적용
216 ApplyRotation();
217 }

256 private void UpdateLightSettings()
257 {
258 // 해의 높이에 따른 밝기 계산 (0 = 수평선, 1 = 정점)
259 float sunHeight = Mathf.Clamp01((Mathf.Sin(currentSunAngle * Mathf.Deg2Rad) + 1f) / 2f);
260 float moonHeight = Mathf.Clamp01((Mathf.Sin(currentMoonAngle * Mathf.Deg2Rad) + 1f) / 2f);
261
262 // 해 빛 설정
263 if (sunLight != null)
264 {
265 // 해가 지평선 위에 있을 때만 빛을 켬
266 bool sunVisible = (currentSunAngle > -10f || currentSunAngle < -170f) && currentSunAngle < 180f;
267 //sunLight.enabled = sunVisible;
268
269 if (sunVisible)
270 {
271 sunLight.intensity = sunIntensityCurve.Evaluate(sunHeight) * maxSunIntensity;
272
273 // 해의 색상 변화 (일출/일몰 시 붉은색)
274 float sunColorFactor = Mathf.Clamp01(sunHeight * 2f); // 낮은 각도에서 붉게
275 sunLight.color = Color.Lerp(new Color(1f, 0.6f, 0.4f), Color.white, sunColorFactor);
276 }
277 else
278 {
279 sunLight.intensity = 0f;
280 }
281
282 if (sunLensFlare != null)
283 {
284 sunLensFlare.intensity = (currentSunAngle > 0f && currentSunAngle < 180f) ? 1.5f : 0f;
285 }
286 }
287
288 // 달 빛 설정
289 if (moonLight != null)
290 {
291 // 달이 지평선 위에 있을 때만 빛을 켬
292 bool moonVisible = currentMoonAngle > 0f && currentMoonAngle < 180f;
293 // moonLight.enabled = moonVisible;
294
295 if (moonVisible)
296 {
297 moonLight.intensity = moonIntensityCurve.Evaluate(moonHeight) * maxMoonIntensity;
298 moonLight.color = new Color(0.8f, 0.8f, 1f); // 차가운 달빛
299 }
300 else
301 {
302 moonLight.intensity = 0f;
303 }
304
305 // LensFlare 컴포넌트의 intensity 설정
306
307 if (moonLensFlare != null)
308 {
309 moonLensFlare.intensity = moonVisible ? 1f : 0f;
310 }

312 }

320 private Vector3 CalculatePosition(float angle)
321 {
322 float radianAngle = angle * Mathf.Deg2Rad;
323
324 // Y축(높이)과 Z축(전후)을 기준으로 한 원형 회전
325 // X축은 좌우 움직임 (원한다면 추가 가능)
326 float x = rotationCenter.x;
327 float y = rotationCenter.y + (rotationRadius * Mathf.Sin(radianAngle));
328 float z = rotationCenter.z + (rotationRadius * Mathf.Cos(radianAngle));
329
330 return new Vector3(x, y, z);
331 }

338 private float NormalizeAngle(float angle)
339 {
340 while (angle > 180f)
341 angle -= 360f;
342 while (angle < -180f)
343 angle += 360f;
344 return angle;
345 }
