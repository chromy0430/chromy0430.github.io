93 private void CheckUnlock(int currentReputation)
94 {
95 List<UnlockableItem> itemsToUnlock = new List<UnlockableItem>();
96
97 foreach (var item in unlockableItems)
98 {
99 if (item.unlockableButton != null && !item.unlockableButton.interactable && currentReputation >= item.reputationThreshold)
100 {
101 itemsToUnlock.Add(item);
102 }
103 }
104
105 // 해금할 아이템들을 처리
106 foreach (var item in itemsToUnlock)
107 {
108 // 버튼 상호작용 활성화
109 item.unlockableButton.interactable = true;
110
111 // 해당 버튼에 연결된 마스크 찾아서 파괴
112 if (instantiatedMasks.TryGetValue(item.unlockableButton, out GameObject maskInstance))
113 {
114 Destroy(maskInstance);
115 instantiatedMasks.Remove(item.unlockableButton); // Dictionary에서 제거
116 }
117
118 Debug.Log($"'{item.itemName}'이(가) 해금되었습니다! (필요 명성도: {item.reputationThreshold})");
119 }
