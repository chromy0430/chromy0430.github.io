113 public List<Vector3Int> CalculatePosition(Vector3Int gridPosition, Vector2Int objectSize, Quaternion rotation, Grid grid)
114 {
115 List<Vector3Int> positions = new List<Vector3Int>();
116
117 // 회전 각도를 0, 90, 180, 270도로 반올림
118 float angle = Mathf.Round(rotation.eulerAngles.y / 90) * 90;
119
120 // 회전된 오브젝트 크기 계산
121 Vector2Int size = objectSize;
122 if (Mathf.Approximately(angle % 180, 90)) // 90도 또는 270도일 때
123 {
124 size = new Vector2Int(objectSize.y, objectSize.x); // 너비와 높이 교환
125 }
126
127 // 회전 각도에 따른 기준점 오프셋 계산
128 Vector2Int offset;
129 if (Mathf.Approximately(angle, 0))
130 {
131 offset = new Vector2Int(0, 0); // 0도: 기준점 그대로
132 }
133 else if (Mathf.Approximately(angle, 90))
134 {
135 offset = new Vector2Int(0, size.y - 1); // 90도: Z축으로 조정
136 }
137 else if (Mathf.Approximately(angle, 180))
138 {
139 offset = new Vector2Int(size.x - 1, size.y - 1); // 180도: X와 Z 모두 조정
140 }
141 else // 270도
142 {
143 offset = new Vector2Int(size.x - 1, 0); // 270도: X축으로 조정
144 }
145
146 // 기준 위치 조정 (왼쪽 상단 코너 기준)
147 Vector3Int startPos = gridPosition - new Vector3Int(offset.x, 0, offset.y);
148
149 // 점유 셀 위치 계산
150 for (int x = 0; x < size.x; x++)
151 {
152 for (int y = 0; y < size.y; y++)
153 {
154 Vector3Int cellPos = startPos + new Vector3Int(x, 0, y);
155 positions.Add(cellPos);
156 }
157 }
158
159 return positions;
160 }

235 public bool CanPlaceObjectAt(Vector3Int gridPosition, Vector2Int objectSize, Quaternion rotation, Grid grid, bool isWall)
236 {
237 List<Vector3Int> positions = CalculatePosition(gridPosition, objectSize, rotation, grid);
238
239 foreach (var pos in positions)
240 {
241 if (placedObjects.ContainsKey(pos) && placedObjects[pos].Count > 0)
242 {
243 // 이 GridData가 관리하는 타입과 같은 타입의 오브젝트만 고려
244 var sameTypeExistingObjects = placedObjects[pos]
245 .AsValueEnumerable().Where(obj => PlacementSystem.Instance.database.GetObjectData(obj.ID).IsWall == isWall);
246 //.Where(obj => PlacementSystem.Instance.database.GetObjectData(obj.ID).IsWall == isWall);
247
248 if (isWall) // 벽 배치 가능 여부 확인 (같은 각도 벽 충돌 체크)
249 {
250 if (sameTypeExistingObjects.Any(obj => Mathf.Approximately(Quaternion.Angle(obj.Rotation, rotation), 0)))
251 {
252 return false; // 같은 각도의 벽이 이미 있음
253 }
254 }
255 else // 가구 배치 가능 여부 확인 (다른 가구 충돌 체크)
256 {
257 if (sameTypeExistingObjects.Any()) // 이미 같은 타입(가구) 오브젝트가 있다면
258 {
259 return false; // 배치 불가
260 }
261 }
262 // --- 다른 타입과의 충돌 검사는 여기서 제거 ---
263 }
264 }
265
266 // 모든 위치 검사를 통과하면 배치 가능
267 return true;
268 }

278 public bool RemoveObjectByIndex(int placedObjectIndex)
279 {
280 bool removed = false;
281 List<Vector3Int> keysToRemove = new List<Vector3Int>();
282
283 foreach (var kvp in placedObjects)
284 {
285 Vector3Int pos = kvp.Key;
286 List<PlacementData> objectsAtPos = kvp.Value;
287
288 // 해당 인덱스의 PlacementData 찾기
289 for (int i = objectsAtPos.Count - 1; i >= 0; i--)
290 {
291 if (objectsAtPos[i].PlacedObjectIndex == placedObjectIndex)
292 {
293 objectsAtPos.RemoveAt(i);
294 removed = true;
295 }
296 }
297
298 // 해당 위치에 더 이상 오브젝트가 없으면 키 제거 준비
299 if (objectsAtPos.Count == 0)
300 {
301 keysToRemove.Add(pos);
302 }
303 }
304
305 // 빈 키 제거
306 foreach (var key in keysToRemove)
307 {
308 placedObjects.Remove(key);
309 }
310
311 return removed;
312 }
