112 public List<Vector3Int> CalculatePosition(Vector3Int gridPosition, Vector2Int objectSize, Quaternion rotation, Grid grid)
113 {
114 List<Vector3Int> positions = new List<Vector3Int>();
115
116 // 회전 각도를 0, 90, 180, 270도로 반올림
117 float angle = Mathf.Round(rotation.eulerAngles.y / 90) * 90;
118
119 // 회전된 오브젝트 크기 계산
120 Vector2Int size = objectSize;
121 if (Mathf.Approximately(angle % 180, 90)) // 90도 또는 270도일 때
122 {
123 size = new Vector2Int(objectSize.y, objectSize.x); // 너비와 높이 교환
124 }
125
126 // 회전 각도에 따른 기준점 오프셋 계산
127 Vector2Int offset;
128 if (Mathf.Approximately(angle, 0))
129 {
130 offset = new Vector2Int(0, 0); // 0도: 기준점 그대로
131 }
132 else if (Mathf.Approximately(angle, 90))
133 {
134 offset = new Vector2Int(0, size.y - 1); // 90도: Z축으로 조정
135 }
136 else if (Mathf.Approximately(angle, 180))
137 {
138 offset = new Vector2Int(size.x - 1, size.y - 1); // 180도: X와 Z 모두 조정
139 }
140 else // 270도
141 {
142 offset = new Vector2Int(size.x - 1, 0); // 270도: X축으로 조정
143 }
144
145 Vector3Int startPos = new Vector3Int(gridPosition.x - offset.x, gridPosition.y, gridPosition.z - offset.y);
146
147 // 점유 셀 위치 계산
148 for (int x = 0; x < size.x; x++)
149 {
150 for (int z = 0; z < size.y; z++) // 루프 변수를 z로 변경하여 명확하게 합니다.
151 {
152 Vector3Int cellPos = new Vector3Int(startPos.x + x, startPos.y, startPos.z + z);
153 positions.Add(cellPos);
154 }
155 }
156
157
158 return positions;
159 }

173 public bool CanPlaceObjectAt(Vector3Int gridPosition, Vector2Int objectSize, Quaternion rotation, Grid grid, bool isWall)
174 {
175 List<Vector3Int> positions = CalculatePosition(gridPosition, objectSize, rotation, grid);
176
177 foreach (var pos in positions)
178 {
179 if (placedObjects.ContainsKey(pos) && placedObjects[pos].Count > 0)
180 {
181 // 이 GridData가 관리하는 타입과 같은 타입의 오브젝트만 고려
182 var sameTypeExistingObjects = placedObjects[pos]
183 .AsValueEnumerable().Where(obj => PlacementSystem.Instance.database.GetObjectData(obj.ID).IsWall == isWall);
184 //.Where(obj => PlacementSystem.Instance.database.GetObjectData(obj.ID).IsWall == isWall);
185
186 if (isWall) // 벽 배치 가능 여부 확인 (같은 각도 벽 충돌 체크)
187 {
188 if (sameTypeExistingObjects.Any(obj => Mathf.Approximately(Quaternion.Angle(obj.Rotation, rotation), 0)))
189 {
190 return false; // 같은 각도의 벽이 이미 있음
191 }
192 }
193 else // 가구 배치 가능 여부 확인 (다른 가구 충돌 체크)
194 {
195 if (sameTypeExistingObjects.Any()) // 이미 같은 타입(가구) 오브젝트가 있다면
196 {
197 return false; // 배치 불가
198 }
199 }
200 // --- 다른 타입과의 충돌 검사는 여기서 제거 ---
201 }
202 }
203
204 // 모든 위치 검사를 통과하면 배치 가능
205 return true;
206 }

216 public bool RemoveObjectByIndex(int placedObjectIndex)
217 {
218 bool removed = false;
219 List<Vector3Int> keysToRemove = new List<Vector3Int>();
220
221 foreach (var kvp in placedObjects)
222 {
223 Vector3Int pos = kvp.Key;
224 List<PlacementData> objectsAtPos = kvp.Value;
225
226 // 해당 인덱스의 PlacementData 찾기
227 for (int i = objectsAtPos.Count - 1; i >= 0; i--)
228 {
229 if (objectsAtPos[i].PlacedObjectIndex == placedObjectIndex)
230 {
231 objectsAtPos.RemoveAt(i);
232 removed = true;
233 }
234 }
235
236 // 해당 위치에 더 이상 오브젝트가 없으면 키 제거 준비
237 if (objectsAtPos.Count == 0)
238 {
239 keysToRemove.Add(pos);
240 }
241 }
242
243 // 빈 키 제거
244 foreach (var key in keysToRemove)
245 {
246 placedObjects.Remove(key);
247 }
248
249 return removed;
250 }
