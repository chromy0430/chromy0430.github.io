255 private RoomInfo FloodFillRoom(Vector3Int startPos, HashSet<Vector3Int> visitedCells)
256 {
257 RoomInfo room = new RoomInfo();
258 Queue<Vector3Int> queue = new Queue<Vector3Int>();
259 queue.Enqueue(startPos);
260 visitedCells.Add(startPos);
261
262 Vector3Int minBounds = startPos;
263 Vector3Int maxBounds = startPos;
264
265 DebugLog($"방 탐색 시작 - 시작 위치: {startPos}");
266
267 // 4방향 탐색
268 Vector3Int[] directions = new Vector3Int[]
269 {
270 new Vector3Int(1, 0, 0), // 오른쪽
271 new Vector3Int(-1, 0, 0), // 왼쪽
272 new Vector3Int(0, 0, 1), // 앞
273 new Vector3Int(0, 0, -1) // 뒤
274 };
275
276 HashSet<Vector3Int> roomCells = new HashSet<Vector3Int>();
277 roomCells.Add(startPos);
278
279 while (queue.Count > 0)
280 {
281 Vector3Int current = queue.Dequeue();
282
283 if (!roomGrid.TryGetValue(current, out RoomCell currentCell))
284 {
285 DebugLog($"셀 없음: {current}");
286 continue;
287 }
288
289 if (currentCell.isFloor)
290 {
291 room.floorCells.Add(current);
292 minBounds = Vector3Int.Min(minBounds, current);
293 maxBounds = Vector3Int.Max(maxBounds, current);
294
295 // 주변 셀 검사
296 foreach (var dir in directions)
297 {
298 Vector3Int neighbor = current + dir;
299
300 // 이웃 셀이 문인지 확인
301 bool isDoorBetween = false;
302 if (roomGrid.TryGetValue(neighbor, out RoomCell neighborCell))
303 {
304 if (neighborCell.isDoor)
305 {
306 isDoorBetween = true;
307 foreach (var obj in neighborCell.objects)
308 {
309 if (!room.doors.Contains(obj))
310 {
311 room.doors.Add(obj);
312 DebugLog($"문 발견: {neighbor}");
313 }
314 }
315 }
316 }
317
318 // 문이 있는 방향으로는 더 이상 진행하지 않음
319 if (!isDoorBetween)
320 {
321 // 벽 확인
322 if (roomGrid.TryGetValue(neighbor, out RoomCell wallCell) && wallCell.isWall)
323 {
324 foreach (var obj in wallCell.objects)
325 {
326 if (!room.walls.Contains(obj))
327 {
328 room.walls.Add(obj);
329 DebugLog($"벽 발견: {neighbor}");
330 }
331 }
332 }
333
334 // 침대 확인
335 if (roomGrid.TryGetValue(neighbor, out RoomCell bedCell) && bedCell.isBed)
336 {
337 foreach (var obj in bedCell.objects)
338 {
339 if (!room.beds.Contains(obj))
340 {
341 room.beds.Add(obj);
342 DebugLog($"침대 발견: {neighbor}");
343 }
344 }
345 }
346
347 // 바닥이 있고 아직 방문하지 않은 경우에만 큐에 추가
348 if (roomGrid.TryGetValue(neighbor, out RoomCell floorCell) &&
349 floorCell.isFloor &&
350 !visitedCells.Contains(neighbor))
351 {
352 queue.Enqueue(neighbor);
353 visitedCells.Add(neighbor);
354 roomCells.Add(neighbor);
355 DebugLog($"다음 바닥 탐색: {neighbor}");
356 }
357 }
358 }
359 }
360 }
361
362 if (room.floorCells.Count > 0)
363 {
364 Vector3 worldMin = grid.GetCellCenterWorld(minBounds);
365 Vector3 worldMax = grid.GetCellCenterWorld(maxBounds);
366 room.bounds = new Bounds();
367 room.bounds.SetMinMax(worldMin, worldMax);
368 room.center = room.bounds.center;
369 room.roomId = $"Room_{room.center.x:F0}_{room.center.z:F0}";
370
371 DebugLog($"방 감지 완료:\n" +
372 $"ID: {room.roomId}\n" +
373 $"중심점: {room.center}\n" +
374 $"바닥: {room.floorCells.Count}개\n" +
375 $"벽: {room.walls.Count}개\n" +
376 $"문: {room.doors.Count}개\n" +
377 $"침대: {room.beds.Count}개\n" +
378 $"유효성: {room.isValid(minWalls, minDoors, minBeds)}");
379
380 return room;
381 }
382
383 DebugLog("유효한 방이 감지되지 않음");
384 return null;
385 }

403 private bool AreRoomsEqual(RoomInfo room1, RoomInfo room2)
404 {
405 if (room1.roomId != room2.roomId || room1.center != room2.center)
406 return false;
407
408 if (room1.walls.Count != room2.walls.Count ||
409 room1.doors.Count != room2.doors.Count ||
410 room1.beds.Count != room2.beds.Count ||
411 room1.floorCells.Count != room2.floorCells.Count)
412 return false;
413
414 bool wallsEqual = room1.walls.AsValueEnumerable().All(w1 => room2.walls.AsValueEnumerable().Any(w2 => w2.GetInstanceID() == w1.GetInstanceID()));
415 bool doorsEqual = room1.doors.AsValueEnumerable().All(d1 => room2.doors.AsValueEnumerable().Any(d2 => d2.GetInstanceID() == d1.GetInstanceID()));
416 bool bedsEqual = room1.beds.AsValueEnumerable().All(b1 => room2.beds.AsValueEnumerable().Any(b2 => b2.GetInstanceID() == b1.GetInstanceID()));
417
418 if (!wallsEqual || !doorsEqual || !bedsEqual)
419 return false;
420
421 var sortedFloors1 = room1.floorCells.AsValueEnumerable().OrderBy(v => v.x).ThenBy(v => v.z).ToList();
422 var sortedFloors2 = room2.floorCells.AsValueEnumerable().OrderBy(v => v.x).ThenBy(v => v.z).ToList();
423
424 for (int i = 0; i < sortedFloors1.Count; i++)
425 {
426 if (sortedFloors1[i] != sortedFloors2[i])
427 return false;
428 }
429
430 return true;
431 }
