90 private void CheckForAvailableQuests(int hour, int minute)
91 {
92 int currentDay = TimeSystem.Instance.CurrentDay;
93
94 // 현재 시간까지 발생했어야 하는 모든 퀘스트를 찾습니다. (ToList()로 복사하여 순회 중 변경 문제 방지)
95 List<QuestData> questsToTrigger = availableQuests.Where(q =>
96 (q.requiredDay < currentDay) ||
97 (q.requiredDay == currentDay && q.requiredHour <= hour && q.requiredMin <= minute)
98 ).ToList();
99
100 foreach (var quest in questsToTrigger)
101 {
102 // 대기열에 추가하고, 이제 확인이 필요 없는 availableQuests에서는 제거합니다.
103 if (!pendingQuests.Contains(quest))
104 {
105 pendingQuests.Enqueue(quest);
106 Debug.Log($"퀘스트 '{quest.questName}'가 대기열에 추가되었습니다.");
107 }
108 availableQuests.Remove(quest);
109 }
110
111 // 대기열에 있는 다음 퀘스트를 표시 시도합니다.
112 TryShowNextQuest();
113 }

157 private void CheckQuestCompletion()
158 {
159 for (int i = activeQuests.Count - 1; i >= 0; i--)
160 {
161 // isCompleted가 true로 바뀌는 순간에만 UI를 업데이트하기 위한 플래그
162 bool justCompleted = false;
163 var quest = activeQuests[i];
164
165 if (quest.isCompleted) continue;
166
167 // 이전에 기록된 진행도를 저장
168 int previousAmount = quest.currentAmount;
169 bool conditionMet = false;
170 switch (quest.data.completionType)
171 {
172 //튜토리얼 퀘스트
173 case QuestCompletionType.Tutorial:
174 conditionMet = true;
175 break;
176
177 case QuestCompletionType.BuildObject:
178 int currentBuildCount = 0;
179 currentBuildCount = CheckBuildObject(quest, currentBuildCount);
180
181 quest.currentAmount = currentBuildCount; // UI 업데이트를 위해 현재 개수 저장
182
183 if (currentBuildCount >= quest.data.completionAmount)
184 {
185 conditionMet = true;
186 }
187 break;
188
189 case QuestCompletionType.EarnMoney:
190 // 퀘스트 수락 후 번 돈 계산
191 int moneyEarned = PlayerWallet.Instance.money - quest.startingMoney;
192 quest.currentAmount = moneyEarned; // UI 표시를 위해 현재 진행도 업데이트
193 if (moneyEarned >= quest.data.completionAmount) conditionMet = true;
194 break;
195
196 case QuestCompletionType.ReachReputation:
197 if (ReputationSystem.Instance.CurrentReputation >= quest.data.completionAmount)
198 {
199 conditionMet = true;
200 }
201 break;
202 }
203
204 // 진행도가 변경되었다면 UI 업데이트 호출
205 if (quest.currentAmount != previousAmount)
206 {
207 questLogUI.UpdateQuestStatus(quest);
208 }
209
210 if (conditionMet && !quest.isCompleted)
211 {
212 quest.isCompleted = true;
213 justCompleted = true;
214 }
215
216 // 방금 막 완료된 경우에만 UI를 한 번 더 업데이트해서 "완료 가능!"으로 바꿈
217 if (justCompleted)
218 {
219 questLogUI.UpdateQuestStatus(quest);
220 }
221 }
222 }

224 private static int CheckBuildObject(ActiveQuest quest, int currentBuildCount)
225 {
226 if (PlacementSystem.Instance != null)
227 {
228 // 중복 계산을 방지하기 위해 이미 센 오브젝트의 인덱스를 저장합니다.
229 HashSet<int> countedObjectIndices = new HashSet<int>();
230
231 // 바닥, 가구, 벽, 장식 등 모든 종류의 그리드 데이터를 순회합니다.
232 GridData[] allGridData = {
233 PlacementSystem.Instance.floorData,
234 PlacementSystem.Instance.furnitureData,
235 PlacementSystem.Instance.wallData,
236 PlacementSystem.Instance.decoData
237 };
238
239 foreach (GridData gridData in allGridData)
240 {
241 if (gridData == null || gridData.placedObjects == null) continue;
242
243 // gridData에 저장된 모든 오브젝트 정보를 확인합니다.
244 foreach (var dataList in gridData.placedObjects.Values)
245 {
246 foreach (var data in dataList)
247 {
248 // 퀘스트 목표 ID와 일치하고, 아직 센 적 없는 오브젝트라면 카운트합니다.
249 if (data.ID == quest.data.completionTargetID && !countedObjectIndices.Contains(data.PlacedObjectIndex))
250 {
251 currentBuildCount++;
252 countedObjectIndices.Add(data.PlacedObjectIndex); // 셌다고 표시
253 }
254 }
255 }
256 }
257 }
258
259 return currentBuildCount;
260 }
